Save lepiter pages to doc directory

Find the location of the lepiter pages. make a doc directory next to it. 

thisPage := thisSnippet parent.
database := thisPage database.
lepiterDirectory := database monitor localRoot.
docDirectory := lepiterDirectory parent / 'doc'.
docDirectory ensureCreateDirectory.
sequencer := LeExportUUIDSequencer new.
pageLinks := LeExportPageLinksBuilder new
		database: database;
		html;
		sequencer: sequencer;
		build;
		links.
database pagesDo: [:page |
	context := LeHtmlContext new
			page: page;
			database: database;
			pageLinks: pageLinks;
			sequencer: sequencer.
	html := LeHtmlPageExporter new
		context: context;
		piece: LeHtmlGtBookPiece gtSinglePage;
		contents.
	(docDirectory / (pageLinks linkForPage: page ifFound: [:link | link fileNameWithExtension] ifNone: ['missing.html'])) asFileReference writeStreamDo: [:w | w nextPutAll: html]]
